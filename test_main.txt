"""
test_main.py — pytest suite for Genomed Assist Pro

Matches the real application exactly:
    GET    /health
    POST   /doctor/register
    POST   /doctor/login
    GET    /doctor/profile
    GET    /doctor/my-patients
    POST   /patient/add
    GET    /patients
    GET    /patient/{patient_id}
    DELETE /patient/{patient_id}
    POST   /appointment/book
    GET    /appointments
    PUT    /appointment/{appointment_id}
    GET    /tutorial
    GET    /tutorial/again
    GET    /help
    GET    /faq

DATABASE ISOLATION
-------------------
`main.py` hardcodes `DATABASE_FILE = "genomed.db"` and calls
`initialize_database()` at import time — there is no env var to redirect
it. To avoid ever touching your real `genomed.db`, this suite:

  1. Sets GENOMED_SECRET_KEY / GENOMED_ENCRYPTION_KEY *before* importing
     `main` (those are read from the environment at import time).
  2. chdir()s into a fresh temporary directory *before* importing `main`,
     so the "genomed.db" created by the module-level
     `initialize_database()` call lands harmlessly in that temp dir.
  3. Monkeypatches `main.DATABASE_FILE = "test_genomed.db"` and calls
     `main.initialize_database()` again so every subsequent `get_db()`
     call (which re-reads the module global at call time) uses a
     distinctly-named, empty test database inside that same temp dir.
  4. On teardown, restores the original working directory and deletes
     the whole temp directory (removing both "genomed.db" and
     "test_genomed.db" — nothing written to disk survives the run).

This requires ZERO changes to main.py.
"""

import os
import shutil
import sys
import tempfile
import uuid
from pathlib import Path

# ---------------------------------------------------------------------------
# 1. ENVIRONMENT + WORKING DIRECTORY SETUP — must happen before `import main`
# ---------------------------------------------------------------------------

_ORIGINAL_CWD = os.getcwd()
_TEST_TMP_DIR = tempfile.mkdtemp(prefix="genomed_test_")

# These are read via os.getenv(...) at module import time in main.py, so
# they must be set before main is imported.
os.environ["GENOMED_SECRET_KEY"] = "test-secret-key-do-not-use-in-prod"
os.environ["GENOMED_ENCRYPTION_KEY"] = Fernet.generate_key().decode()

os.chdir(_TEST_TMP_DIR)

import main  # noqa: E402  (must be imported after env vars + chdir above)

TEST_DB_FILE = "test_genomed.db"
main.DATABASE_FILE = TEST_DB_FILE  # redirect all future get_db() calls
main.initialize_database()  # create fresh tables in the isolated test db

class TestDataIsolationBetweenDoctors:
    """Patients/appointments are scoped by doctor_id — verify one doctor
    cannot see or modify another doctor's records."""

    @pytest.fixture(scope="class")
    def second_doctor_headers(self, client):
        creds = {
            "username": _unique_username("second"),
            "password": "OtherSecure!",
            "name": "Dr. Marie Curie",
            "specialization": "Radiology",
        }
        reg = client.post("/doctor/register", json=creds)
        assert reg.status_code == 201
        login = client.post(
            "/doctor/login", json={"username": creds["username"], "password": creds["password"]}
        )
        assert login.status_code == 200
        token = login.json()["access_token"]
        return {"Authorization": f"Bearer {token}"}

    def test_other_doctor_cannot_see_patient(self, client, second_doctor_headers, created_patient):
        response = client.get(f"/patient/{created_patient['id']}", headers=second_doctor_headers)
        assert response.status_code == 404

    def test_other_doctor_patients_list_excludes_patient(self, client, second_doctor_headers, created_patient):
        response = client.get("/patients", headers=second_doctor_headers)
        assert response.status_code == 200
        ids = [p["id"] for p in response.json()["patients"]]
        assert created_patient["id"] not in ids
